void Receive(uint8_t* Buf, uint32_t Len){
	int j=0;
	if(Buf[0]=='S' && Buf[1]=='M' && Buf[2]=='S' && Buf[Len-3]=='E' && Buf[Len-2]=='N' && Buf[Len-1]=='D'){
		for(j=0;j<Len;j++){
			validez+=Buf[j];
		}
		validez=validez-Buf[Len-4];
		if (validez==Buf[Len-4]){
			ccr_llegada=(Buf[4]<<8)|(Buf[5]);
			CDC_Transmit_FS("OK", strlen(OK));
			validez=0;
		}else{
			CDC_Transmit_FS("NO", strlen(OK));
		}
	}

}

static void ProtocoloQT(uint16_t rpms, uint16_t adc_read, uint8_t * Buf){

	uint8_t tam=12,s=83,m=77,e=69,n=78,d=68,val=0,b1_rpms=0,b2_rpms=0,b1_adc=0,b2_adc=0;

	if(rpms==0){

		b1_rpms=0;
		b2_rpms=0;
	}
	if(adc_read==0){

		b1_adc=0;
		b2_adc=0;
	}


	b1_rpms=rpms>>8;
	b2_rpms=rpms;
	b1_adc=adc_read>>8;
	b2_adc=adc_read;

	val=s+m+s+tam+b1_rpms+b2_rpms+b1_adc+b2_adc+val+e+n+d;

	Buf[0]=s;
	Buf[1]=m;
	Buf[2]=s;
	Buf[3]=tam;
	Buf[4]=b1_rpms;
	Buf[5]=b2_rpms;
	Buf[6]=b1_adc;
	Buf[7]=b2_adc;
	Buf[8]=val;
	Buf[9]=e;
	Buf[10]=n;
	Buf[11]=d;

}
